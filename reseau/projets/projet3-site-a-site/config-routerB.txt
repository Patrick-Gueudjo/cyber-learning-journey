! ========================
! PROJET 3 – RouterB
! Site B – Réseau 192.168.30.0/24
! Lien WAN vers RouterA – 10.10.10.0/30
! ========================

enable
configure terminal

! Nom du routeur (facultatif)
hostname RouterB

! Interface LAN – vers PC-B / Switch
interface GigabitEthernet0/0
 description LAN Site B
 ip address 192.168.30.1 255.255.255.0
 no shutdown

! Interface WAN – vers RouterA
interface Serial0/0/0
 description Lien WAN vers RouterA
 ip address 10.10.10.2 255.255.255.252
 no shutdown

! Routage statique vers le LAN du Site A
ip route 192.168.10.0 255.255.255.0 10.0.0.1

end
write memory
